API Keys
Chưa có API key nào.
Ví dụ sử dụng
curl -X POST https://api.legalai.vn/v1/legal/ask \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "question": "Thời hạn thử việc tối đa là bao nhiêu?"
  }'
import requests

headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}

response = requests.post(
    "https://api.legalai.vn/v1/legal/ask",
    headers=headers,
    json={"question": "Thời hạn thử việc tối đa?"}
)

print(response.json())
const response = await fetch('https://api.legalai.vn/v1/legal/ask', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    question: 'Thời hạn thử việc tối đa?'
  })
});

const data = await response.json();
console.log(data);
